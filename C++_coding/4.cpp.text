//write a c++ function to count the number of words in a sentence (string)?:




#include <iostream>
#include <string.h>

using namespace std;

int main()
{
    char x[100];
    int count = 0 , i;
    gets(x);
    for (i=0; x[i] != '\0';i++)
    {
         if ( x[i] == ' ')
            count++;
    }
    cout << " Number of words in sentence are: " << count + 1;
    return 0;
}
