//write a c++ program to find the factorial of a given number using a recursive function?:



#include <iostream>
using namespace std;

 int factorial( int n) {
    if (n == 0) {
        return 1;
    } else {
        return n * factorial(n - 1);
    }
}

int main() {
    int n;
    cout << "Enter a Number: ";
    cin >> n;
    cout << "!" << n << " = " << factorial(n);
    return 0;
}
