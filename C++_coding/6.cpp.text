#implement c++ program to find the bigest and smallest element in array:?










#include <iostream>

using namespace std;

int main()
{
    int n;
    int arr[n];
    for (int i=1;i<n;i++){
        cin >> arr[i];
    }
    int max= arr[0];
    int min= arr[0];
    for(int i=1;i<n;i++){
        if(arr[i]<min){
            max = arr[i];
        }
        if(arr[i]<max){
            min = arr[i];
        }
    }
    cout << max;
    cout << min;
    return 0;
}
